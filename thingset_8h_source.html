56 /*
57  * ThingSet addressing in 29-bit CAN ID similar to SAE J1939
58  *
59  * In order to avoid collisions with SAE J1939 and NMEA 2000, bit 25 (EDP) is always set to 1 and
60  * bit 24 (DP) is used to distinguish between request/response and pub/sub messages.
61  *
62  * Request/response messages using ISO-TP:
63  *
64  * --------------------------------------------------------------------
65  * | 28 .. 26 | 25 | 24 | 23 .. 16 | 15 .. 8 | 7 .. 0 |
66  * --------------------------------------------------------------------
67  * | Priority | 1 | 0 | reserved (0xDA) | target addr | source addr |
68  * --------------------------------------------------------------------
69  *
70  * Bits 23..16 are set to 218 (0xDA) as suggested by ISO-TP standard (ISO 15765-2) for normal
71  * fixed addressing with N_TAtype = physical
72  *
73  * Control and pub/sub messages (always single-frame):
74  *
75  * --------------------------------------------------------------------
76  * | 28 .. 26 | 25 | 24 | 23 .. 16 | 15 .. 8 | 7 .. 0 |
77  * --------------------------------------------------------------------
78  * | Priority | 1 | 1 | data ID (MSB) | data ID (LSB) | source addr |
79  * --------------------------------------------------------------------
80  *
81  * Priority:
82  * 0 .. 3: High-priority control frames
83  * 4 .. 7: Normal pub/sub frames for monitoring
84  */

238 #define TS_ROLE_USR (1U << 0) // normal user
239 #define TS_ROLE_EXP (1U << 1) // expert user
240 #define TS_ROLE_MKR (1U << 2) // maker

242 #define TS_READ_MASK 0x00FF // read flags stored in 4 least-significant bits
243 #define TS_WRITE_MASK 0xFF00 // write flags stored in 4 most-significant bits
244 
245 #define TS_USR_MASK (TS_ROLE_USR << 8 | TS_ROLE_USR)
246 #define TS_EXP_MASK (TS_ROLE_EXP << 8 | TS_ROLE_EXP)
247 #define TS_MKR_MASK (TS_ROLE_MKR << 8 | TS_ROLE_MKR)
248 
249 #define TS_READ(roles) ((roles) & TS_READ_MASK)
250 #define TS_WRITE(roles) (((roles) << 8) & TS_WRITE_MASK)
251 #define TS_READ_WRITE(roles) (TS_READ(roles) | TS_WRITE(roles))
